24  class FileSuite extends Suite {
25      private static final ParametersRunnerFactory DEFAULT_FACTORY = new BlockJUnit4ClassRunnerWithParametersFactory();
26      private static final List<Runner> NO_RUNNERS = Collections.emptyList();

28      private final List<Runner> runners;
29      private final List<Object> params;

31      /**
32       * @param klass InnerParameterizedTest.class
33       * @param test  a test with parameters containing a file
34       * @throws Throwable
35       */
36      public FileSuite(Class<?> klass, TestWithParameters test) throws Throwable {
37          super(klass, NO_RUNNERS);
38  
39          Parameterized.Parameters parameters = this.getParametersMethod().getAnnotation(Parameterized.Parameters.class);
40  
41          params = test.getParameters();
42  
43          final File file = new File(params.get(0).toString());
44          if (!file.exists()) throw new RuntimeException(format("A file {0} does not exists in test parameters.", file));
45  
46          List<Object> primaryKeyList = getPrimaryKeyList(file);
47  
48          ParametersRunnerFactory runnerFactory = getParametersRunnerFactory(klass);
49  
50          runners = Collections.unmodifiableList(
51              createRunnersForParameters(primaryKeyList, parameters.name(), runnerFactory));
52      }

54      private static <T> List<T> copyIterator(Iterator<T> iter) {
55          List<T> copy = new ArrayList<>();
56          while (iter.hasNext())
57              copy.add(iter.next());
58          return copy;
59      }

61      private static List<Object> getPrimaryKeyList(File file)
62          throws IOException, InvalidFormatException {
63          List<Object> primaryKeyList = new LinkedList<>();
64  
65          Workbook wb = WorkbookFactory.create(file);
66          Sheet sheet = wb.getSheet("Tabelle1");
67  
68          if (sheet == null) {
69              String msg = MessageFormat.format("No sheet \"{0}\" in workbook {1}.", "Tabelle1", file);
70              throw new RuntimeException(msg);
71          }
72          for (int rownum = sheet.getFirstRowNum() + 1; rownum < sheet.getLastRowNum() + 1; rownum++) {
73              Row row = sheet.getRow(rownum);
74              Cell cell = row.getCell(0, Row.CREATE_NULL_AS_BLANK);
75              if (cell.getStringCellValue().isEmpty()) continue;
76              String id = cell.getStringCellValue();
77  
78              primaryKeyList.add(new Object[]{id, copyIterator(row.iterator())});
79          }
80          return primaryKeyList;
81      }

83      private static String removeExtension(String filename) {
84          if (filename == null) {
85              return null;
86          }
87          int index = filename.lastIndexOf(".");
88          if (index == -1) {
89              return filename;
90          } else {
91              return filename.substring(0, index);
92          }
93      }

95      private static TestWithParameters createTestWithParameters(TestClass testClass, String pattern, int index, Object[] parameters) {
96          String finalPattern = pattern.replaceAll("\\{index\\}", Integer.toString(index));
97          String name = MessageFormat.format(finalPattern, parameters);
98          return new TestWithParameters("[" + name + "]", testClass, Arrays.asList(parameters));
99      }

101     protected String getName() {
102         File file = new File(params.get(0).toString());
103         String fName = removeExtension(file.getName());
104         String className = getTestClass().getName().substring(getTestClass().getName().lastIndexOf(".") + 1);
105         return MessageFormat.format("{0}->\"{1}\"", className, fName);
106     }

108     private ParametersRunnerFactory getParametersRunnerFactory(AnnotatedElement klass)
109         throws InstantiationException, IllegalAccessException {
110         UseParametersRunnerFactory annotation =
111             klass.getAnnotation(UseParametersRunnerFactory.class);
112         if (annotation == null) {
113             return DEFAULT_FACTORY;
114         } else {
115             Class factoryClass = annotation.value();
116             return (ParametersRunnerFactory) factoryClass.newInstance();
117         }
118     }

120     private List<Runner> createRunnersForParameters(Iterable<Object> allParameters, String namePattern, ParametersRunnerFactory runnerFactory) throws Exception {
121         try {
122             List list = this.createTestsForParameters(allParameters, namePattern);
123             List<Runner> runners = new ArrayList<>();
124 
125             for (Object anE : list) {
126                 TestWithParameters test = (TestWithParameters) anE;
127                 runners.add(runnerFactory.createRunnerForTestWithParameters(test));
128             }
129 
130             return runners;
131         } catch (ClassCastException var8) {
132             throw this.parametersMethodReturnedWrongType();
133         }
134     }
135 
136     private FrameworkMethod getParametersMethod() throws Exception {
137         List methods = this.getTestClass().getAnnotatedMethods(Parameterized.Parameters.class);
138         Iterator it = methods.iterator();
139 
140         FrameworkMethod each;
141         do {
142             if (!it.hasNext()) {
143                 throw new Exception("No public static parameters method on class " + this.getTestClass().getName());
144             }
145 
146             each = (FrameworkMethod) it.next();
147         } while (!each.isStatic() || !each.isPublic());
148 
149         return each;
150     }

152     private TestWithParameters createTestWithNotNormalizedParameters(String pattern, int index, Object parametersOrSingleParameter) {
153         Object[] parameters = parametersOrSingleParameter instanceof Object[] ? (Object[]) parametersOrSingleParameter : new Object[]{parametersOrSingleParameter};
154         return createTestWithParameters(this.getTestClass(), pattern, index, parameters);
155     }

157     private Exception parametersMethodReturnedWrongType() throws Exception {
158         String className = this.getTestClass().getName();
159         String methodName = this.getParametersMethod().getName();
160         String message = MessageFormat.format("{0}.{1}() must return an Iterable of arrays.", className, methodName);
161         return new Exception(message);
162     }

164     private List<TestWithParameters> createTestsForParameters(Iterable<Object> allParameters,
165         String namePattern) {
166         int i = 0;
167         List<TestWithParameters> children = new ArrayList<>();
168 
169         for (Object parametersOfSingleTest : allParameters) {
170             children.add(this.createTestWithNotNormalizedParameters(namePattern, i++, parametersOfSingleTest));
171         }
172 
173         return children;
174     }

181     /**
182      * The <code>SuiteClass</code> annotation specifies the classes to be run when a class
183      * annotated with <code>@RunWith(Suite.class)</code> is run.
184      */
185     @Retention(RetentionPolicy.RUNTIME)
186     @Target(ElementType.TYPE)
187     @Inherited
188     public @interface SuiteClass {
189         /**
190          * @return the classes to be run
191          */
192         Class<?> value();
